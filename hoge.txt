
    GameField* field = GameSystem::getField();
    int hp(0);
    for(auto it = field->begin(); it != field->end(); ++it)
    {
        if((*it)->getType() == PLAYER)
        {
            hp = (*it)->getHP();
        }
    }

    std::string msg("HP: ");
    for(int i = 0; i < hp/10; ++i)
    {
        msg += "#";
    }
    this->print(msg, 90, 4);

    return;
